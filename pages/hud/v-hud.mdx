## Installation

### Prerequisites
- **Node.js 18+**: Required for building the interface
- **QBCore Framework**: Latest version recommended
- **Modern Browser**: Chrome 90+, Firefox 88+, or equivalent
- **Build Tools**: Git, npm/pnpm/yarn

### Development Installation

1. **Clone Repository**
   ```bash
   cd resources/[hud]
   git clone https://github.com/example/v-hud.git
   cd v-hud
   ```

2. **Install Dependencies**
   ```bash
   # Using npm
   npm install
   
   # Using pnpm (recommended)
   pnpm install
   
   # Using yarn
   yarn install
   ```

3. **Build for Production**
   ```bash
   # Development build with hot reload
   npm run dev
   
   # Production build
   npm run build
   
   # Preview production build
   npm run preview
   ```

4. **Server Configuration**
   ```cfg
   # Add to server.cfg
   start v-hud
   ```

### Production Installation

1. **Download Release**
   ```bash
   # Download pre-built release
   wget https://github.com/example/v-hud/releases/latest/download/v-hud.zip
   unzip v-hud.zip -d resources/[hud]/
   ```

2. **Configure Server**
   ```cfg
   ensure v-hud
   ```

## Component Architecture

### Core Components

```typescript
// components/HUD/HUD.tsx
import React, { useCallback, useMemo } from 'react'
import { useHUDStore } from '@/stores/hudStore'
import { HealthBar } from './HealthBar'
import { VehicleHUD } from './VehicleHUD'
import { NotificationSystem } from './NotificationSystem'

export const HUD: React.FC = () => {
  const { 
    isVisible, 
    playerData, 
    vehicleData, 
    theme,
    elements 
  } = useHUDStore()
  
  const visibleElements = useMemo(() => 
    Object.entries(elements).filter(([, config]) => config.enabled),
    [elements]
  )
  
  if (!isVisible) return null
  
  return (
    <div className={`hud-container theme-${theme}`}>
      {visibleElements.map(([key, config]) => {
        switch (key) {
          case 'health':
            return <HealthBar key={key} config={config} value={playerData.health} />
          case 'vehicle':
            return <VehicleHUD key={key} config={config} data={vehicleData} />
          default:
            return null
        }
      })}
      
      <NotificationSystem />
    </div>
  )
}
```

### Health Bar Component

```typescript
// components/HUD/HealthBar.tsx
import React from 'react'
import { motion, AnimatePresence } from 'framer-motion'
import { useTheme } from '@/hooks/useTheme'

interface HealthBarProps {
  value: number
  config: ElementConfig
}

export const HealthBar: React.FC<HealthBarProps> = ({ value, config }) => {
  const { colors } = useTheme()
  
  const percentage = Math.max(0, Math.min(100, value))
  const isLow = percentage < 25
  const isCritical = percentage < 10
  
  return (
    <motion.div
      className="health-bar"
      style={{
        position: 'absolute',
        left: config.position.x,
        bottom: Math.abs(config.position.y),
        width: config.size?.width || 200,
        height: config.size?.height || 8
      }}
      initial={{ opacity: 0, y: 20 }}
      animate={{ opacity: 1, y: 0 }}
      transition={{ duration: 0.3 }}
    >
      <div className="bar-background">
        <motion.div
          className={`bar-fill ${isLow ? 'low' : ''} ${isCritical ? 'critical' : ''}`}
          style={{
            backgroundColor: isCritical 
              ? colors.error 
              : isLow 
                ? colors.warning 
                : colors.health
          }}
          initial={{ width: 0 }}
          animate={{ width: `${percentage}%` }}
          transition={{ duration: 0.5, ease: 'easeOut' }}
        />
      </div>
      
      {(isLow || isCritical) && (
        <motion.div
          className="warning-pulse"
          animate={{ 
            opacity: [0.5, 1, 0.5],
            scale: [1, 1.05, 1]
          }}
          transition={{ 
            duration: 1,
            repeat: Infinity,
            ease: 'easeInOut'
          }}
        />
      )}
    </motion.div>
  )
}
```

### State Management

```typescript
// stores/hudStore.ts
import { create } from 'zustand'
import { subscribeWithSelector } from 'zustand/middleware'

interface HUDState {
  // Visibility
  isVisible: boolean
  
  // Player data
  playerData: {
    health: number
    armor: number
    hunger: number
    thirst: number
    stress: number
    money: {
      cash: number
      bank: number
    }
  }
  
  // Vehicle data
  vehicleData: {
    speed: number
    fuel: number
    engine: number
    gear: number
    inVehicle: boolean
  }
  
  // UI state
  theme: string
  elements: Record<string, ElementConfig>
  notifications: Notification[]
  
  // Actions
  updatePlayerData: (data: Partial<HUDState['playerData']>) => void
  updateVehicleData: (data: Partial<HUDState['vehicleData']>) => void
  setTheme: (theme: string) => void
  toggleVisibility: () => void
  addNotification: (notification: Omit<Notification, 'id'>) => void
  removeNotification: (id: string) => void
}

export const useHUDStore = create<HUDState>()(
  subscribeWithSelector((set, get) => ({
    // Initial state
    isVisible: true,
    
    playerData: {
      health: 100,
      armor: 0,
      hunger: 100,
      thirst: 100,
      stress: 0,
      money: { cash: 0, bank: 0 }
    },
    
    vehicleData: {
      speed: 0,
      fuel: 100,
      engine: 1000,
      gear: 1,
      inVehicle: false
    },
    
    theme: 'modern',
    elements: elements,
    notifications: [],
    
    // Actions
    updatePlayerData: (data) => 
      set(state => ({
        playerData: { ...state.playerData, ...data }
      })),
      
    updateVehicleData: (data) =>
      set(state => ({
        vehicleData: { ...state.vehicleData, ...data }
      })),
      
    setTheme: (theme) => set({ theme }),
    
    toggleVisibility: () => 
      set(state => ({ isVisible: !state.isVisible })),
      
    addNotification: (notification) => 
      set(state => ({
        notifications: [
          ...state.notifications,
          { ...notification, id: crypto.randomUUID() }
        ]
      })),
      
    removeNotification: (id) =>
      set(state => ({
        notifications: state.notifications.filter(n => n.id !== id)
      }))
  }))
)
```

// Memoization and optimization
const OptimizedHealthBar = React.memo(HealthBar, (prevProps, nextProps) => {
  return prevProps.value === nextProps.value && 
         prevProps.config === nextProps.config

// Custom hooks for performance
const useThrottledValue = (value: number, delay: number) => {
  const [throttledValue, setThrottledValue] = useState(value)
  
  useEffect(() => {
    const timer = setTimeout(() => setThrottledValue(value), delay)
    return () => clearTimeout(timer)
  }, [value, delay])
  
  return throttledValue
}

// Virtual scrolling for notifications
