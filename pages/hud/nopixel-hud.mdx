## Installation

### Prerequisites
- QBCore Framework or ESX
- Modern web browser support
- FiveM server with proper permissions

### Quick Setup

1. **Download Resource**
   ```bash
   cd resources/[hud]
   git clone https://github.com/example/nopixel-hud.git
   ```

2. **Install to Server**
   ```cfg
   # Add to server.cfg
   start nopixel-hud
   ```

3. **Configuration**
   ```lua
   -- config.lua
   Config.Framework = 'qb-core'  -- or 'esx'
   Config.EnableAllFeatures = true
   ```

## Configuration

### Basic Settings

```lua
Config = {}

-- Framework Detection
Config.Framework = 'qb-core'        -- 'qb-core', 'esx', 'standalone'
Config.Debug = false                -- Enable debug mode

-- HUD Elements
Config.Elements = {
    health = true,
    armor = true,
    food = true,
    water = true,
    stress = true,          -- QBCore only
    stamina = false,        -- Optional
    oxygen = true,          -- When underwater
    
    -- Vehicle elements
    speedometer = true,
    fuel = true,
    engine = true,
    seatbelt = true,
    
    -- UI elements
    cash = true,
    job = true,
    time = true,
    location = false        -- Street name display
}

-- Position Settings
Config.Position = {
    health = {x = 30, y = -220},
    armor = {x = 30, y = -190},
    food = {x = 30, y = -160},
    water = {x = 30, y = -130},
    cash = {x = 30, y = -50},
    speedometer = {x = -350, y = -150}
}
```

## Commands & Usage

### Player Commands

```lua
-- HUD Controls
/hud                    -- Toggle HUD visibility
/nphud                  -- NoPixel HUD specific toggle
/hud position           -- Adjust element positions
/hud reset              -- Reset to default positions

-- Information Display
/time                   -- Show current time
/cash                   -- Display money (if enabled)
/job                    -- Show current job info
```

### Admin Commands

```lua
-- HUD Administration
/hudsettings            -- Open HUD configuration panel
/sethudelements [player]-- Configure elements for player
/hudreload              -- Reload HUD configuration

-- Department Management
/setbadge [id] [badge]  -- Set police badge number
/setcallsign [id] [sign]-- Set radio callsign
/setrank [id] [rank]    -- Set department rank
```

### Multi-Framework Support

```lua
-- Framework detection and adaptation
if Config.Framework == 'qb-core' then
    QBCore = exports['qb-core']:GetCoreObject()
    
    -- QBCore specific events
    RegisterNetEvent('QBCore:Client:OnPlayerLoaded', function()
        TriggerEvent('nopixel-hud:client:initialize')
    end)
    
elseif Config.Framework == 'esx' then
    ESX = exports['es_extended']:getSharedObject()
    
    -- ESX specific events  
    RegisterNetEvent('esx:playerLoaded', function()
        TriggerEvent('nopixel-hud:client:initialize')
    end)
end
```

## API Reference

### Client Events

```lua
-- Update player status
TriggerEvent('nopixel-hud:client:updateHealth', health)
TriggerEvent('nopixel-hud:client:updateArmor', armor)
TriggerEvent('nopixel-hud:client:updateFood', food)
TriggerEvent('nopixel-hud:client:updateWater', water)

-- Department/Job updates
TriggerEvent('nopixel-hud:client:updateJob', jobData)
TriggerEvent('nopixel-hud:client:setBadge', badgeNumber)
TriggerEvent('nopixel-hud:client:setCallsign', callsign)

-- Vehicle updates
TriggerEvent('nopixel-hud:client:updateSpeed', speed)
TriggerEvent('nopixel-hud:client:updateFuel', fuel)
TriggerEvent('nopixel-hud:client:updateEngine', health)
```

### Exports

```lua
-- Get HUD data
local hudVisible = exports['nopixel-hud']:IsVisible()
local currentJob = exports['nopixel-hud']:GetCurrentJob()

-- Control HUD
exports['nopixel-hud']:ShowElement('health', true)
exports['nopixel-hud']:HideElement('armor', false)
exports['nopixel-hud']:SetTheme('police')

-- Notifications
exports['nopixel-hud']:ShowNotification({
    type = 'police',
    title = 'LSPD Alert',
    message = '10-54 in progress',
    duration = 8000
})
```
