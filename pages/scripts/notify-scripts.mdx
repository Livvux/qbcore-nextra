## Installation

### Quick Setup

1. **Download Notify Scripts**
   ```bash
   cd resources/[qb]
   git clone https://github.com/example/notify-scripts.git
   ```

2. **Install Dependencies**
   ```bash
   cd notify-scripts
   npm install        # If using build process
   ```

3. **Server Configuration**
   ```cfg
   # Add to server.cfg
   start notify-scripts
   
   # Optional: Replace default QBCore notifications
   setr qb-core:useNotifyScript "notify-scripts"
   ```

## Usage Examples

### Basic Notifications

```lua
-- Client-side usage
exports['notify-scripts']:Notify('Success!', 'Your action was completed successfully.', 'success', 5000)

exports['notify-scripts']:Notify('Error!', 'Something went wrong.', 'error', 8000)

exports['notify-scripts']:Notify('Warning!', 'Please check your input.', 'warning', 6000)

exports['notify-scripts']:Notify('Info', 'This is an informational message.', 'info', 4000)

-- Server-side usage
TriggerClientEvent('notify-scripts:client:notify', playerId, {
    title = 'Server Message',
    message = 'Welcome to the server!',
    type = 'success',
    duration = 5000
})
```

### Advanced Notifications

```lua
    type = 'info',

```

### QBCore Integration

```lua
-- Replace default QBCore notifications
if GetResourceState('notify-scripts') == 'started' then
    -- Override QBCore notify function
    RegisterNetEvent('QBCore:Notify', function(text, type, length)
        exports['notify-scripts']:Notify('Notification', text, type or 'info', length or 5000)
    end)
    
    -- Override with advanced options
    RegisterNetEvent('QBCore:NotifyAdvanced', function(data)
        exports['notify-scripts']:NotifyAdvanced(data)
    end)
end

-- Server-side integration
local function QBNotify(source, text, type, length)
    if GetResourceState('notify-scripts') == 'started' then
        TriggerClientEvent('notify-scripts:client:notify', source, {
            title = 'Server',
            message = text,
            type = type or 'info',
            duration = length or 5000
        })
    else
        TriggerClientEvent('QBCore:Notify', source, text, type, length)
    end
end

-- Usage in other resources
QBNotify(playerId, 'Welcome to the server!', 'success', 8000)
```

### JavaScript API (NUI)

```javascript
// Send notification from NUI
window.postMessage({
    action: 'notify',
    data: {
        title: 'NUI Notification',
        message: 'Sent from NUI interface',
        type: 'info',
        duration: 5000
    }
}, '*');

// Handle notification actions
window.addEventListener('message', (event) => {
    if (event.data.action === 'notificationAction') {
        const { actionId, notificationId } = event.data;
        
        // Handle the action
        fetch(`https://${GetParentResourceName()}/handleNotificationAction`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify({
                action: actionId,
                notificationId: notificationId
            })
        });
    }
});
```

### Event System

```lua
-- Client Events
RegisterNetEvent('notify-scripts:client:notify', function(data)
    exports['notify-scripts']:Notify(data.title, data.message, data.type, data.duration)
end)

RegisterNetEvent('notify-scripts:client:notifyAdvanced', function(data)
    exports['notify-scripts']:NotifyAdvanced(data)
end)

RegisterNetEvent('notify-scripts:client:clear', function()
    exports['notify-scripts']:ClearNotifications()
end)

-- Server Events  
RegisterNetEvent('notify-scripts:server:notificationAction', function(data)
    local src = source
    
    -- Handle notification actions
    if data.action == 'accept_invite' then
        -- Handle accept action
        TriggerClientEvent('party:client:acceptInvite', src)
    elseif data.action == 'decline_invite' then
        -- Handle decline action
        TriggerClientEvent('party:client:declineInvite', src)
    end
end)
```

## Performance Tips

### Optimization

```lua
-- Limit notifications per second to prevent spam
local notificationCooldown = {}

function CanShowNotification(source)
    local currentTime = GetGameTimer()
    local lastNotification = notificationCooldown[source] or 0
    
    if currentTime - lastNotification > 500 then -- 500ms cooldown
        notificationCooldown[source] = currentTime
        return true
    end
    
    return false
end

-- Batch notifications to reduce network calls
local notificationBatch = {}

CreateThread(function()
    while true do
        if next(notificationBatch) then
            for playerId, notifications in pairs(notificationBatch) do
                TriggerClientEvent('notify-scripts:client:batchNotify', playerId, notifications)
            end
            notificationBatch = {}
        end
        
        Wait(100) -- Send batches every 100ms
    end
end)

-- Add to batch instead of sending immediately
function QueueNotification(playerId, notification)
    if not notificationBatch[playerId] then
        notificationBatch[playerId] = {}
    end
    
    table.insert(notificationBatch[playerId], notification)
end
```
